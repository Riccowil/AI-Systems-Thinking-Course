class AgentComponent(BaseModel):
    id: str          # min 1, max 50
    name: str        # min 1, max 200
    component_type: Literal["agent", "tool", "memory", "evaluator", "constraint"]
    description: str # optional, max 500
    position_x: Optional[float]
    position_y: Optional[float]

class AgentLink(BaseModel):
    from_component: str   # min 1, max 50
    to_component: str     # min 1, max 50
    polarity: Polarity    # reuse existing enum
    strength: Optional[float]  # 0.0-1.0, default 1.0
    label: str            # optional, max 200
    link_type: Optional[Literal["data_flow", "control_flow", "feedback", "dependency"]]

    def to_causal_link(self) -> CausalLink:
        return CausalLink(
            from_var=self.from_component,
            to_var=self.to_component,
            polarity=self.polarity,
            strength=self.strength
        )
